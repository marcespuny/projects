-- How many entries in the database are from Africa?

SELECT COUNT() FROM countries
WHERE continent LIKE 'Africa';

56

-- What was the total population of Oceania in 2005?

SELECT SUM(population_years.population) FROM countries
JOIN population_years
ON countries.id = population_years.country_id
WHERE countries.continent LIKE 'Oceania' AND population_years.year LIKE 2005;


32.66417

-- What is the average population of countries in South America in 2003?

SELECT AVG(population) FROM countries
LEFT JOIN population_years
ON countries.id = population_years.country_id
WHERE continent LIKE 'South America' AND year LIKE 2003;

25.8906514285714

-- What country had the smallest population in 2007?

GROUP BY continent;
